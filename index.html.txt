<script>
(() => {
  const $ = (id) => document.getElementById(id);

  const state = {
    players: [], // {name, score}
    turn: 0,
    dartsThisTurn: 0,
    turnStartScore: 501,
    mult: 1,
    base: 20,
    lastActions: [] // stack for undo
  };

  const msg = (t) => { $("msg").textContent = t; };

  function renderPlayers() {
    const root = $("players");
    root.innerHTML = "";
    state.players.forEach((p, idx) => {
      const div = document.createElement("div");
      div.className = "card" + (idx === state.turn ? " turn" : "");
      div.innerHTML = `
        <div class="pname">${p.name}</div>
        <div class="score">${p.score}</div>
        <div class="tag">${idx === state.turn ? "Your throw" : "Waiting"}</div>
      `;
      root.appendChild(div);
    });

    $("turnLabel").textContent = state.players.length
      ? `Turn: ${state.players[state.turn].name}`
      : "Turn: —";
  }

  function updateThrowUI(label) {
    $("multLabel").textContent = label;
    $("throwValue").textContent = state.base * state.mult;
  }

  function setMult(m, label) {
    state.mult = m;
    updateThrowUI(label);
    // highlight not needed, but could be added
  }

  function clampTurn() {
    if (!state.players.length) { state.turn = 0; return; }
    state.turn = (state.turn + state.players.length) % state.players.length;
  }

  function startTurn() {
    if (!state.players.length) return;
    state.dartsThisTurn = 0;
    state.turnStartScore = state.players[state.turn].score;
    $("dartCount").textContent = state.dartsThisTurn;
    msg(`${state.players[state.turn].name} to throw. Score: ${state.players[state.turn].score}`);
  }

  function nextPlayer() {
    if (!state.players.length) return;
    state.turn++;
    clampTurn();
    startTurn();
    renderPlayers();
  }

  function isCheckoutLegal(newScore, wasDouble) {
    // Must finish exactly on 0 and final dart must be a double (GOAL ring)
    if (newScore === 0 && !wasDouble) return false;
    return true;
  }

  function addThrow(points, wasDouble=false, label="") {
    if (!state.players.length) return msg("Add at least 1 player.");

    const p = state.players[state.turn];

    // Lock to 3 darts per turn
    if (state.dartsThisTurn >= 3) {
      msg("3 darts used — hit Next.");
      return;
    }

    const prevScore = p.score;
    const attempted = prevScore - points;

    // Bust conditions:
    // - score goes below 0
    // - score hits 0 but not on double
    // - score becomes 1 (standard darts rule because can't finish on double)
    const bust =
      attempted < 0 ||
      attempted === 1 ||
      (attempted === 0 && !wasDouble);

    // Save for undo
    state.lastActions.push({
      turn: state.turn,
      prevScore,
      prevDarts: state.dartsThisTurn,
      prevTurnStartScore: state.turnStartScore,
      didBust: bust
    });
    $("undoBtn").disabled = state.lastActions.length === 0;

    if (bust) {
      // revert to turn start score and consume remaining darts (end turn)
      p.score = state.turnStartScore;
      state.dartsThisTurn = 3;
      $("dartCount").textContent = state.dartsThisTurn;
      renderPlayers();
      msg(`BUST! “FREE KICK CARLTON.” Back to ${state.turnStartScore}. Hit Next.`);
      return;
    }

    // Apply score
    p.score = attempted;
    state.dartsThisTurn++;
    $("dartCount").textContent = state.dartsThisTurn;
    renderPlayers();

    // Optional AFL callout
    if (label === "SUPER GOAL" && (points === 60 || points === 57)) {
      msg(`💥 50!!! (optional house rule) — ${p.name} scored ${points}.`);
    } else {
      msg(`${p.name} scored ${points} ${label ? `(${label})` : ""}. Remaining: ${p.score}`);
    }

    if (p.score === 0) {
      msg(`🏆 ${p.name} WINS! Finished on a GOAL (double).`);
    }
  }

  function undo() {
    const a = state.lastActions.pop();
    $("undoBtn").disabled = state.lastActions.length === 0;
    if (!a) return;

    state.turn = a.turn;
    const p = state.players[state.turn];
    p.score = a.prevScore;
    state.dartsThisTurn = a.prevDarts;
    state.turnStartScore = a.prevTurnStartScore;

    $("dartCount").textContent = state.dartsThisTurn;
    renderPlayers();
    msg("Undid last throw.");
  }

  function resetGame() {
    state.players.forEach(p => p.score = 501);
    state.turn = 0;
    state.dartsThisTurn = 0;
    state.turnStartScore = 501;
    state.lastActions = [];
    $("undoBtn").disabled = true;
    $("dartCount").textContent = "0";
    renderPlayers();
    msg("Game reset. 501 each. Add throw to start.");
    startTurn();
  }

  // UI bindings
  $("addBtn").addEventListener("click", () => {
    const name = $("nameInput").value.trim();
    if (!name) return;
    state.players.push({ name, score: 501 });
    $("nameInput").value = "";
    clampTurn();
    renderPlayers();
    if (state.players.length === 1) startTurn();
    msg(`Added ${name}.`);
  });

  $("nameInput").addEventListener("keydown", (e) => {
    if (e.key === "Enter") $("addBtn").click();
  });

  $("baseSelect").addEventListener("change", (e) => {
    state.base = Number(e.target.value);
    $("throwValue").textContent = state.base * state.mult;
  });

  $("singleBtn").addEventListener("click", () => setMult(1, "Single"));
  $("doubleBtn").addEventListener("click", () => setMult(2, "GOAL"));
  $("tripleBtn").addEventListener("click", () => setMult(3, "SUPER GOAL"));

  $("obullBtn").addEventListener("click", () => addThrow(25, false, "BEHIND"));
  $("ibullBtn").addEventListener("click", () => addThrow(50, false, "GOAL SQUARE"));

  $("scoreBtn").addEventListener("click", () => {
    const points = state.base * state.mult;
    const wasDouble = state.mult === 2;
    const label = state.mult === 1 ? "Single" : state.mult === 2 ? "GOAL" : "SUPER GOAL";
    addThrow(points, wasDouble, label);
  });

  $("undoBtn").addEventListener("click", undo);
  $("resetBtn").addEventListener("click", resetGame);
  $("nextBtn").addEventListener("click", nextPlayer);

  // init
  renderPlayers();
  updateThrowUI("Single");
})();
</script>
